function plot() {
    local cmd=("$@")
    o="${orgFile:h}/${imgdir:-images}/$(md5m "$cmd[*]").png"
    mkdir -p "${o:h}"
    print -r -- $c | reval "$cmd[@]" > $o || { echo Exited $? ; return 1 }
    print -nr -- "./$(realpath $o --relative-to ${orgFile:h})"
}

function plot-basic() {
    a="${1:?}"
    b="${2:-$a}"
    plot calendarheatmap -maxcount 0 -colorscale $a -colorscalealt $b
}
